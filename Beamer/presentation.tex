  \end{theorem}
  \begin{block}{Proof Sketch}
      \begin{enumerate}
      \item Every simple polygon can be guarded 
            with at most $\lfloor n/3\rfloor$ vertex guards.
      \item Some simple polygons require at least $\lfloor n/3\rfloor$ guards.
      \end{enumerate}
  \end{block}
